\section{ARM architecture}

\begin{frame}{von Neumann architecture}
\end{frame}

\begin{frame}[fragile]{CISC}
\begin{columns}
\column{0.4\textwidth}
\begin{verbatim}
export fn foo() u32 {
    var x: u32 = 42;
    var y: u32 = 23;
    return x + y;
}
\end{verbatim}

\column{0.6\textwidth}
\begin{verbatim}
...
mov     dword ptr [rbp - 8], 42
mov     dword ptr [rbp - 12], 23
mov     eax, dword ptr [rbp - 8]
add     eax, dword ptr [rbp - 12]
...
\end{verbatim}
\end{columns}

\vspace{1cm}

\url{https://zig.godbolt.org/z/qoMfEK}
\end{frame}

\begin{frame}[fragile]{RISC}
\begin{columns}
\column{0.4\textwidth}
\begin{verbatim}
export fn foo() u32 {
    var x: u32 = 42;
    var y: u32 = 23;
    return x + y;
}
\end{verbatim}

\column{0.6\textwidth}
\begin{verbatim}
...
movw    r0, #42
str     r0, [sp, #8]
movw    r0, #23
str     r0, [sp, #4]
ldr     r0, [sp, #8]
ldr     r1, [sp, #4]
adds    r0, r0, r1
...
\end{verbatim}
\end{columns}

\vspace{1cm}

\url{https://zig.godbolt.org/z/Mee9ro}
\end{frame}

\begin{frame}[fragile]{ARM instruction set}
\begin{itemize}
    \item<1-> Data Processing
    \begin{verbatim}
    add r0, r1, #2\end{verbatim}
    \item<2-> Memory
    \begin{verbatim}
    ldr r0, [sp, #4]\end{verbatim}
    \item<3-> Branching
    \begin{verbatim}
    b label
    bx lr\end{verbatim}
\end{itemize}
\end{frame}
